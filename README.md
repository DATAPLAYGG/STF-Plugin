# STF-Plugin
A Minecraft plugin to enhance shift-click behavior Paper1.20.1


commands:
  stfset:
    description: Set the amount of durability increase
    usage: /stfset <amount>
    permission: shifttofix.setdurability

  stf:
    description: Help command
    usage: /stf
    permission: stf.help

  stfreload:
    description: Reload the plugin configuration
    usage: /stfreload
    permission: shifttofix.reload

  stfon:
    description: Enable ShiftToFix plugin functionality
    usage: /stfon
    permission: shifttofix.toggle
